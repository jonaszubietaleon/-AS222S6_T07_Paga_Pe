# Etapa 1: Build del proyecto
FROM node:20-alpine AS builder

WORKDIR /app
COPY package*.json ./
RUN npm ci --omit=dev
COPY . .
RUN npm run build

# Etapa 2: Producción ultraligera
FROM pierrezemb/mini-web-server

# Copiar archivos estáticos al directorio por defecto
COPY --from=builder /app/build /srv/http

EXPOSE 80
